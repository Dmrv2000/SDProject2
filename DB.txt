CREATE TABLE utilizadores
(
  u_id INT NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),
  u_name VARCHAR(100) NOT NULL,
  u_pwd VARCHAR(100) NOT NULL
);

CREATE TABLE produtos
(
  p_id INT NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),
  p_name VARCHAR(100) NOT NULL,
  p_stock INT NOT NULL,
  p_precoVenda FLOAT NOT NULL,
  P_precoCompra FLOAT NOT NULL,
  p_categoria VARCHAR(100) NOT NULL
);

CREATE TABLE administradores
(
  u_id INT NOT NULL PRIMARY KEY,
  FOREIGN KEY (u_id) REFERENCES utilizadores(u_id) ON DELETE CASCADE
);

CREATE TABLE vendas
(
  v_id INT NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),
  v_quant INT NOT NULL,
  p_id INT NOT NULL,
  v_data DATE NOT NULL,
  v_precoTotal FLOAT NOT NULL,
  FOREIGN KEY (p_id) REFERENCES produtos(p_id) ON DELETE CASCADE
);
